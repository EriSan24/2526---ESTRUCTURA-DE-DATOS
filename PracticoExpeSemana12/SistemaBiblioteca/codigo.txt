using System;
using System.Collections.Generic;

class Biblioteca
{
    static void Main(string[] args)
    {
        Dictionary<int, string> libros = new Dictionary<int, string>();
        int opcion;

        do
        {
            Console.WriteLine("=== SISTEMA DE BIBLIOTECA ===");
            Console.WriteLine("1. Registrar libro");
            Console.WriteLine("2. Consultar libro");
            Console.WriteLine("3. Mostrar todos los libros");
            Console.WriteLine("4. Salir");
            Console.Write("Seleccione una opcion: ");
            opcion = int.Parse(Console.ReadLine());

            switch (opcion)
            {
                case 1:
                    Console.Write("Ingrese codigo del libro: ");
                    int codigo = int.Parse(Console.ReadLine());

                    Console.Write("Ingrese titulo y autor: ");
                    string info = Console.ReadLine();

                    libros[codigo] = info;
                    Console.WriteLine("Libro registrado correctamente.");
                    break;

                case 2:
                    Console.Write("Ingrese codigo del libro: ");
                    int buscar = int.Parse(Console.ReadLine());

                    if (libros.ContainsKey(buscar))
                    {
                        Console.WriteLine("Libro: " + libros[buscar]);
                    }
                    else
                    {
                        Console.WriteLine("Libro no encontrado.");
                    }
                    break;

                case 3:
                    Console.WriteLine("Lista de libros:");
                    foreach (var libro in libros)
                    {
                        Console.WriteLine("Codigo: " + libro.Key + " - " + libro.Value);
                    }
                    break;
            }

        } while (opcion != 4);
    }
}
